$(document).ready(function () {
            // 10. 在body裡面加一顆按鈕，名稱叫做放大 
            $('body').append('<button>放大</button>');
            
            // 20. 按鈕點選後，把p標籤的字型放大
            // 30. 點選一次，增加2px，慢慢放大
            $('button').on('click', function () {
                // 取的p標籤的字的大小
                // 字的大小放大2px(+2px)
                $('p').animate({fontSize: "+=2px" });
            })
        });
